# loading组件

<template>
    <div class='loading-container' v-show='showLoading'>
        <div>
            <img src="./loading.gif" alt="">
            <span>正在加载</span>
        </div>
    </div>
</template>

<script>
export default {
    data() {
        return {
            showLoading:false
        }
    },
    methods:{
        show(){
            this.showLoading = true;
        },
        close(){
            this.showLoading = false;
        }
    }
}
</script>
